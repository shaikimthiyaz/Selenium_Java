# Webtable sorting

```
package demo;

import java.util.List;
import java.util.stream.Collectors;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.Assert;

public class WebTableSorting {

	public static void main(String[] args) {

		WebDriver driver = new ChromeDriver();
		driver.get("https://rahulshettyacademy.com/seleniumPractise/#/offers");
		driver.manage().window().maximize();
//		click on column name
		driver.findElement(By.xpath("//tr//th[1]")).click();
		
//		collect the all webelements into list
		
		List<WebElement> elements =driver.findElements(By.xpath("//tr//td[1]"));
		
//		capture text of all webelements into list
		
		List<String> originalText =elements.stream().map(s->s.getText()).collect(Collectors.toList());
		System.out.println("original text is"+ originalText);
		
//		now sort the originallist
		List<String> sortedList = originalText.stream().sorted().collect(Collectors.toList());
		System.out.println("sorted text is"+ sortedList);
		
//		now compare original text and sorted list
		
		Assert.assertEquals(originalText, sortedList);

		
	}
	

}
```


# pagination and getting price in a webtable

```
package demo;

import java.util.List;
import java.util.stream.Collectors;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class PaginationAndGetPrice {

	public static void main(String[] args) throws InterruptedException {

		WebDriver driver = new ChromeDriver();
		driver.get("https://rahulshettyacademy.com/seleniumPractise/#/offers");
		driver.manage().window().maximize();
//		click on column name which gives in alphabetic order

		driver.findElement(By.xpath("//tr//th[1]")).click();

		List<String> price;
		do {

//		print the price of the rice
			List<WebElement> elements = driver.findElements(By.xpath("//tr//td[1]"));
			price = elements.stream().filter(e -> e.getText().contains("Rice")).map(e -> getPriceofRice(e))
					.collect(Collectors.toList());
			price.forEach(e -> System.out.println(e));
			if (price.size() < 1) {
				Thread.sleep(1000);
				driver.findElement(By.xpath("//a[@aria-label = 'Next']")).click();

			}

		} while (price.size() < 1);
	}

	public static String getPriceofRice(WebElement e) {
		String pricevalue = e.findElement(By.xpath("//following-sibling::td[1]")).getText();
		return pricevalue;

	}

}
```

# search results filtering

```
package demo;

import java.util.List;
import java.util.stream.Collectors;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.Assert;

public class StreamFilter {

	public static void main(String[] args) {
		WebDriver driver = new ChromeDriver();
		driver.get("https://rahulshettyacademy.com/seleniumPractise/#/offers");
		driver.manage().window().maximize();
		driver.findElement(By.id("search-field")).sendKeys("apple");
		List<WebElement>  fruits=driver.findElements(By.xpath("//tr/td[1]"));
		System.out.println(fruits);
		List<WebElement> results= fruits.stream().filter(e->e.getText().trim().toLowerCase().contains("apple")).collect(Collectors.toList());
		results.forEach(e->System.out.println(e.getText()));
		Assert.assertEquals(fruits.size(), results.size());
	}

}
```
